0 # Contador

1 # Valor de incremento

0
1
0

10 # Inicio do vetor da sequência

.TEXT
# Define local que o dado será escrito
LOAD M(1)
ADD M(6)

# Escreve a na posição calculada
STOR M(102,8:19)
LOAD M(3)
STOR M(0)

# Faz a avançar na sequencia
STOR M(5) # salva a em c
ADD M(4)  # soma a com b
STOR M(3) # salva a
LOAD M(5) # carrega c
STOR M(4) # salva c em b

# Incrementa contador
LOAD M(1)
ADD M(2)
STOR M(1)

# Volta pro inicio se contador < quantidade de números
LOAD M(0)
SUB M(1)
JUMP+ M(100,0:19)
